032import com.pi4j.io.gpio.Pin;
033
034/**
035 * Invalid pin mode exception.
036 *
037 * @author Robert Savage (<a
038 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
039 */
040@SuppressWarnings("unused")
041public class InvalidPinModeException extends RuntimeException {
042
043    private static final long serialVersionUID = -5101222911651959182L;
044    private final Pin pin;
045
046    public InvalidPinModeException(Pin pin, String message) {
047        super(message);
048        this.pin = pin;
049    }
050
051    public Pin getPin() {
052        return pin;
053    }
054}
