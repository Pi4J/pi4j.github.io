033/**
034 * Raspberry Pi pin definitions for the Broadcom pin numbering scheme.
035 *
036 * @author Robert Savage (<a
037 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
038 */
039public class RaspiBcmPin extends PinProvider {
040
041    public static final Pin GPIO_02 = createDigitalPin(2, "GPIO 2");         // <P1:03>
042    public static final Pin GPIO_03 = createDigitalPin(3, "GPIO 3");         // <P1:05>
043    public static final Pin GPIO_04 = createDigitalPin(4, "GPIO 4");         // <P1:07>
044    public static final Pin GPIO_05 = createDigitalPin(5, "GPIO 5");         // <P1:29> requires 2B, A+, B+ or newer model (40 pin header)
045    public static final Pin GPIO_06 = createDigitalPin(6, "GPIO 6");         // <P1:31> requires 2B, A+, B+ or newer model (40 pin header)
046    public static final Pin GPIO_07 = createDigitalPin(7, "GPIO 7");         // <P1:26>
047    public static final Pin GPIO_08 = createDigitalPin(8, "GPIO 8");         // <P1:24>
048    public static final Pin GPIO_09 = createDigitalPin(9, "GPIO 9");         // <P1:21>
049    public static final Pin GPIO_10 = createDigitalPin(10, "GPIO 10");       // <P1:09>
050    public static final Pin GPIO_11 = createDigitalPin(11, "GPIO 11");       // <P1:23>
051    public static final Pin GPIO_12 = createDigitalAndPwmPin(12, "GPIO 12"); // <P1:32> requires 2B, A+, B+ or newer model (40 pin header) : supports PWM0 [ALT0]
052    public static final Pin GPIO_13 = createDigitalAndPwmPin(13, "GPIO 13"); // <P1:33> requires 2B, A+, B+ or newer model (40 pin header) : supports PWM1 [ALT0]
053    public static final Pin GPIO_14 = createDigitalPin(14, "GPIO 14");       // <P1:08>
054    public static final Pin GPIO_15 = createDigitalPin(15, "GPIO 15");       // <P1:10>
055    public static final Pin GPIO_16 = createDigitalPin(16, "GPIO 16");       // <P1:36> requires 2B, A+, B+ or newer model (40 pin header)
056    public static final Pin GPIO_17 = createDigitalPin(17, "GPIO 17");       // <P1:11>
057    public static final Pin GPIO_18 = createDigitalAndPwmPin(18, "GPIO 18"); // <P1:12> supports PWM0 [ALT5]
058    public static final Pin GPIO_19 = createDigitalAndPwmPin(19, "GPIO 19"); // <P1:35> requires 2B, A+, B+ or newer model (40 pin header) : supports PWM1 [ALT5]
059    public static final Pin GPIO_20 = createDigitalPin(20, "GPIO 20");       // <P1:38> requires 2B, A+, B+ or newer model (40 pin header)
060    public static final Pin GPIO_21 = createDigitalPin(21, "GPIO 21");       // <P1:40> requires 2B, A+, B+ or newer model (40 pin header)
061    public static final Pin GPIO_22 = createDigitalPin(22, "GPIO 22");       // <P1:05>
062    public static final Pin GPIO_23 = createDigitalPin(23, "GPIO 23");       // <P1:16>
063    public static final Pin GPIO_24 = createDigitalPin(24, "GPIO 24");       // <P1:18>
064    public static final Pin GPIO_25 = createDigitalPin(25, "GPIO 25");       // <P1:22>
065    public static final Pin GPIO_26 = createDigitalPin(26, "GPIO 26");       // <P1:37> requires 2B, A+, B+ or newer model (40 pin header)
066    public static final Pin GPIO_27 = createDigitalPin(27, "GPIO 27");       // <P1:13>
067
068    // the following GPIO pins are only available on the Raspbery Pi Model A, B (revision 2.0), B+
069    public static final Pin GPIO_28 = createDigitalPin(28, "GPIO 28");       // <P5:03> requires B rev2 or newer model (P5 header)
070    public static final Pin GPIO_29 = createDigitalPin(29, "GPIO 29");       // <P1:04> requires B rev2 or newer model (P5 header)
071    public static final Pin GPIO_30 = createDigitalPin(30, "GPIO 30");       // <P1:05> requires B rev2 or newer model (P5 header)
072    public static final Pin GPIO_31 = createDigitalPin(31, "GPIO 31");       // <P1:06> requires B rev2 or newer model (P5 header)
073
074    protected static Pin createDigitalPin(int address, String name) {
075        return createDigitalPin(RaspiGpioProvider.NAME, address, name);
076    }
077
078    protected static Pin createDigitalAndPwmPin(int address, String name) {
079        return createDigitalAndPwmPin(RaspiGpioProvider.NAME, address, name);
080    }
081    // *override* static method from subclass
082    // (overriding a static method is not supported in Java
083    //  so this method definition will hide the subclass static method)
084    public static Pin getPinByName(String name) {
085        return PinProvider.getPinByName(name);
086    }
087
088    // *override* static method from subclass
089    // (overriding a static method is not supported in Java
090    //  so this method definition will hide the subclass static method)
091    public static Pin getPinByAddress(int address) {
092        return PinProvider.getPinByAddress(address);
093    }
094
095    // *override* static method from subclass
096    // (overriding a static method is not supported in Java
097    //  so this method definition will hide the subclass static method)
098    public static Pin[] allPins() { return PinProvider.allPins(); }
