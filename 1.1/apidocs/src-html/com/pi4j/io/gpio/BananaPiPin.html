033import java.util.EnumSet;
034
035/**
036 * BananaPi pin definitions for (default) WiringPi pin numbering scheme.
037 *
038 * @author Robert Savage (<a
039 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
040 */
041public class BananaPiPin extends PinProvider {
042
043    public static final Pin GPIO_00 = createDigitalPin(0, "GPIO 0");
044    public static final Pin GPIO_01 = createDigitalPin(1, "GPIO 1");
045    public static final Pin GPIO_02 = createDigitalPin(2, "GPIO 2");
046    public static final Pin GPIO_03 = createDigitalPin(3, "GPIO 3");
047    public static final Pin GPIO_04 = createDigitalPin(4, "GPIO 4");
048    public static final Pin GPIO_05 = createDigitalPin(5, "GPIO 5");
049    public static final Pin GPIO_06 = createDigitalPin(6, "GPIO 6");
050    public static final Pin GPIO_07 = createDigitalAndPwmPinNoEdge(7, "GPIO 7");
051    public static final Pin GPIO_08 = createDigitalPinNoEdge(8, "GPIO 8", EnumSet.of(PinPullResistance.OFF, PinPullResistance.PULL_UP));
052    public static final Pin GPIO_09 = createDigitalPinNoEdge(9, "GPIO 9", EnumSet.of(PinPullResistance.OFF, PinPullResistance.PULL_UP));
053    public static final Pin GPIO_10 = createDigitalPin(10, "GPIO 10");
054    public static final Pin GPIO_11 = createDigitalPin(11, "GPIO 11");
055    public static final Pin GPIO_12 = createDigitalPin(12, "GPIO 12");
056    public static final Pin GPIO_13 = createDigitalPin(13, "GPIO 13");
057    public static final Pin GPIO_14 = createDigitalPin(14, "GPIO 14");
058    public static final Pin GPIO_15 = createDigitalPin(15, "GPIO 15");
059    public static final Pin GPIO_16 = createDigitalPin(16, "GPIO 16");
060    public static final Pin GPIO_17 = createDigitalPin(17, "GPIO 17");
061    public static final Pin GPIO_18 = createDigitalPinNoEdge(18, "GPIO 18");
062    public static final Pin GPIO_19 = createDigitalPin(19, "GPIO 19");
063    public static final Pin GPIO_20 = createDigitalPinNoEdge(20, "GPIO 20");
064
065    protected static Pin createDigitalPin(int address, String name) {
066        return createDigitalPin(BananaPiGpioProvider.NAME, address, name);
067    }
068
069    protected static Pin createDigitalPinNoEdge(int address, String name, EnumSet<PinPullResistance> resistance) {
070        return createDigitalPin(BananaPiGpioProvider.NAME, address, name, resistance, EnumSet.noneOf(PinEdge.class));
071    }
072
073    protected static Pin createDigitalPinNoEdge(int address, String name) {
074        return createDigitalPin(BananaPiGpioProvider.NAME, address, name, EnumSet.noneOf(PinEdge.class));
075    }
076
077    protected static Pin createDigitalAndPwmPin(int address, String name) {
078        return createDigitalAndPwmPin(BananaPiGpioProvider.NAME, address, name);
079    }
080
081    protected static Pin createDigitalAndPwmPinNoEdge(int address, String name) {
082        return createDigitalAndPwmPin(BananaPiGpioProvider.NAME, address, name, EnumSet.noneOf(PinEdge.class));
083    }
084
085    // *override* static method from subclass
086    // (overriding a static method is not supported in Java
087    //  so this method definition will hide the subclass static method)
088    public static Pin getPinByName(String name) {
089        return PinProvider.getPinByName(name);
090    }
091
092    // *override* static method from subclass
093    // (overriding a static method is not supported in Java
094    //  so this method definition will hide the subclass static method)
095    public static Pin getPinByAddress(int address) {
096        return PinProvider.getPinByAddress(address);
097    }
098
099    // *override* static method from subclass
100    // (overriding a static method is not supported in Java
101    //  so this method definition will hide the subclass static method)
102    public static Pin[] allPins() { return PinProvider.allPins(); }
103
104    // *override* static method from subclass
105    // (overriding a static method is not supported in Java
106    //  so this method definition will hide the subclass static method)
107    public static Pin[] allPins(PinMode ... mode) { return PinProvider.allPins(mode); }
108}
