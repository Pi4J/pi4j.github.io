034/**
035 * This interface describes a pin.
036 *
037 * @author Robert Savage (<a
038 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
039 */
040@SuppressWarnings("unused")
041public interface Pin extends Comparable<Pin> {
042
043    String getProvider();
044    int getAddress();
045    String getName();
046    EnumSet<PinMode> getSupportedPinModes();
047    boolean supportsPinPullResistance();
048    EnumSet<PinPullResistance> getSupportedPinPullResistance();
049    boolean supportsPinEdges();
050    boolean supportsPinEvents();
051    EnumSet<PinEdge> getSupportedPinEdges();
052}
