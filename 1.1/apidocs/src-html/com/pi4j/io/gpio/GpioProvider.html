034/**
035 * Gpio provider interface.
036 *
037 * @author Robert Savage (<a
038 *         href="http://www.savagehomeautomation.com">http://www.savagehomeautomation.com</a>)
039 */
040@SuppressWarnings("unused")
041public interface GpioProvider {
042
043    String getName();
044
045    boolean hasPin(Pin pin);
046
047    void export(Pin pin, PinMode mode, PinState defaultState);
048    void export(Pin pin, PinMode mode);
049    boolean isExported(Pin pin);
050    void unexport(Pin pin);
051
052    void setMode(Pin pin, PinMode mode);
053    PinMode getMode(Pin pin);
054
055    void setPullResistance(Pin pin, PinPullResistance resistance);
056    PinPullResistance getPullResistance(Pin pin);
057
058    void setState(Pin pin, PinState state);
059    PinState getState(Pin pin);
060
061    void setValue(Pin pin, double value);
062    double getValue(Pin pin);
063
064    void setPwm(Pin pin, int value);
065    void setPwmRange(Pin pin, int range);
066    int getPwm(Pin pin);
067
068    void addListener(Pin pin, PinListener listener);
069    void removeListener(Pin pin, PinListener listener);
070    void removeAllListeners();
071
072    void shutdown();
073    boolean isShutdown();
074}
