032import com.pi4j.io.i2c.I2CFactory.UnsupportedBusNumberException;
033
034public class I2CFactoryProviderBananaPi extends I2CProviderImpl {
035
036    @Override
037    protected String getFilenameForBusnumber(int busNumber) throws UnsupportedBusNumberException {
038        if ((busNumber < 0) || (busNumber > 3 && busNumber < 10) || (busNumber > 17)) {
039            throw new UnsupportedBusNumberException();
040        }
041
042        return "/dev/i2c-" + busNumber;
043    }
044}
