032import java.io.File;
033import java.io.IOException;
034import java.nio.file.Files;
035
036/**
037 * @author Peter Schuebl
038 */
039public abstract class W1BaseDevice implements W1Device {
040
041    private final String id;
042
043    private String name;
044
045    private File deviceDir;
046
047    @Override
048    public String getName() {
049        return name;
050    }
051
052    @Override
053    public String getId() {
054        return id;
055    }
056
057    public W1BaseDevice(final File deviceDir) {
058        String deviceName;
059        try {
060            deviceName = new String(Files.readAllBytes(new File(deviceDir, "name").toPath()));
061        } catch (IOException e) {
062            // FIXME logging
063            deviceName = deviceDir.getName();
064        }
065        name = deviceName;
066        id = deviceName;
067        this.deviceDir = deviceDir;
068    }
069
070    @Override
071    public boolean equals(Object obj) {
072        return obj instanceof W1Device && id.equals(((W1Device) obj).getId());
073    }
074
075    @Override
076    public int hashCode() {
077        return id.hashCode();
078    }
079
080    @Override
081    public String getValue() throws IOException {
082        return new String(Files.readAllBytes(new File(deviceDir, "w1_slave").toPath()));
083    }
084}
