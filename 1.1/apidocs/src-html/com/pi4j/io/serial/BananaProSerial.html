032/**
033 * This class simply exposed the available UART/serial port
034 * address (device file paths) that are exposed on the BananaPro.
035 */
036public class BananaProSerial {
037
038    // (UART0)
039    // this com port is exposed on the J15 header and is typically
040    // configured for terminal access to the operating system
041    // - TX is located at J15-03
042    // - RX is located at J15-02
043    public static final String UART0_COM_PORT = "/dev/ttyS0";
044    public static final String J15_COM_PORT = UART0_COM_PORT;
045
046    // (UART2)
047    // this com port is exposed on the CON6 header
048    // - TX is located at CON6-13
049    // - RX is located at CON6-11
050    public static final String UART2_COM_PORT = "/dev/ttyS1";
051
052    // (UART4)
053    // this is the default com port exposed on the CON6 header
054    // - TX is located at CON6-08
055    // - RX is located at CON6-10
056    public static final String UART4_COM_PORT = "/dev/ttyS2";
057    public static final String DEFAULT_COM_PORT = UART4_COM_PORT;
058
059//    // (UART5)  --- !! NOT CURRENTLY SUPPORTED IN OS !!
060//    // this is the default com port exposed on the CON6 header
061//    // - TX is located at CON6-24
062//    // - RX is located at CON6-23
063//    public static final String UART5_COM_PORT = "/dev/ttyS*";
064
065//    // (UART6)  --- !! NOT CURRENTLY SUPPORTED IN OS !!
066//    // this is the default com port exposed on the CON6 header
067//    // - TX is located at CON6-19
068//    // - RX is located at CON6-21
069//    public static final String UART6_COM_PORT = "/dev/ttyS*";
070
071    // (UART7)
072    // this com port is exposed on the CON6 header
073    // - TX is located at CON6-32
074    // - RX is located at CON6-31
075    public static final String UART7_COM_PORT = "/dev/ttyS3";
076}
